138// Создать обратный рейс
139Flight Flight::createReturnFlight() const {
140 // Создаём обратный рейс с теми же параметрами, но поменяв местами аэропорты
141 Flight returnFlight;
142 returnFlight.setFlightNumber(getReturnFlightNumber());
143 returnFlight.setDepartureAirport(destinationAirport);
144 returnFlight.setDestinationAirport(departureAirport);
145 returnFlight.setAircraftId(aircraftId);
146
147 // Время обратного рейса рассчитываем как время прибытия + время на подготовку
148 // (например, 2 часа на подготовку самолёта)
149 std::time_t preparationTime = 2 * 60 * 60; // 2 часа в секундах
150 returnFlight.setDepartureTime(arrivalTime + preparationTime);
151
152 // Время прибытия обратного рейса рассчитываем как время отправления + продолжительность полёта
153 std::time_t flightDuration = getFlightDuration();
154 returnFlight.setArrivalTime(returnFlight.getDepartureTime() + flightDuration);
155
156 return returnFlight;
157}

175 oss << "Flight #" << flightNumber << std::endl;
176 oss << " Route: " << departureAirport << " -> " << destinationAirport << std::endl;
177 oss << " Aircraft: " << aircraftId << std::endl;

179 // Форматируем время отправления

182 std::tm tmStruct;
183 localtime_s(&tmStruct, &departureTime);
184 std::strftime(depTimeStr, sizeof(depTimeStr), "%Y-%m-%d %H:%M:%S", &tmStruct);
185 oss << " Departure: " << depTimeStr << std::endl;

187
188 // Форматируем время прибытия
189 if (arrivalTime > 0) {
190 char arrTimeStr[100];
191 std::tm tmStruct;
192 localtime_s(&tmStruct, &arrivalTime);
193 std::strftime(arrTimeStr, sizeof(arrTimeStr), "%Y-%m-%d %H:%M:%S", &tmStruct);
194 oss << " Arrival: " << arrTimeStr << std::endl;
195 }
196
197 oss << " Duration: " << std::fixed << std::setprecision(2) << getFlightDurationHours() << " hours" << std::endl;
198 oss << " Status: ";
199 if (completed) {
200 oss << "Completed";
201 } else if (isInProgress()) {
202 oss << "In Progress";
203 } else if (isScheduled()) {

213// Проверить корректность данных рейса
214bool Flight::isValid() const {
215 // Проверяем, что все обязательные поля заполнены
216 if (flightNumber.empty() || departureAirport.empty() ||
217 destinationAirport.empty() || aircraftId.empty()) {
218 return false;
219 }
220
221 // Проверяем, что аэропорты отправления и назначения разные
222 if (departureAirport == destinationAirport) {
223 return false;
224 }
225
226 // Проверяем, что время прибытия больше времени отправления
227 if (arrivalTime <= departureTime) {
228 return false;
229 }
230
231 return true;
232}
