206std::vector<std::string> Schedule::getOverworkedAircraft() const {
207 std::vector<std::string> overworkedAircraft;
208 const double MAX_DAILY_FLIGHT_TIME = 12.0; // Максимум 12 часов в день
209
210 // Группируем рейсы по самолётам и дням
211 std::map<std::string, std::map<std::string, double> > aircraftDailyTime;
212
213 for (const auto& flight : flights) {
214 if (!flight) continue;
215
216 std::string aircraftId = flight->getAircraftId();
217 std::time_t depTime = flight->getDepartureTime();
218
219 // Получаем дату (без времени)
220 std::tm tm;
221 localtime_s(&tm, &depTime);
222 std::string date = std::to_string(tm.tm_year + 1900) + "-" +
223 std::to_string(tm.tm_mon + 1) + "-" +
224 std::to_string(tm.tm_mday);
225
226 aircraftDailyTime[aircraftId][date] += flight->getFlightDurationHours();
227 }
228
229 // Проверяем, какие самолёты перегружены
230 for (const auto& aircraft : aircraftDailyTime) {
231 for (const auto& day : aircraft.second) {
232 if (day.second > MAX_DAILY_FLIGHT_TIME) {
233 overworkedAircraft.push_back(aircraft.first);
234 break;
235 }
236 }
237 }

239 return overworkedAircraft;
